CREATE TABLE Experiments (
    EXP_ID NUMBER NOT NULL,
    PARTICIPANT_ID NUMBER NOT NULL,
	STAGE NUMBER,
	TIME TIMESTAMP,
	TUTORIAL_TIME NUMBER,
	QUIZ_TIME NUMBER,
	RESPONSE_TIME NUMBER,
	CONSISTENT BIT,
	SENARIO_NAME VARCHAR2 (200),
    PRIMARY KEY(EXP_ID,PARTICIPANT_ID,SENARIO,STAGE),
	CONSTRAINT fk_SENARIO_EXP FOREIGN KEY (SENARIO_NAME) REFERENCES SENARIOES(SENARIO_NAME)

);
/
CREATE TABLE SENARIOES (
	SENARIO_NAME VARCHAR2 (200) NOT NULL,
    PRIMARY KEY(SENARIO_NAME)
);
/
CREATE TABLE ITEMS (
	ITEM_NAME VARCHAR2 (200) NOT NULL,
    PRIMARY KEY(ITEM_NAME)
);
/
CREATE TABLE GROUPS (
	GROUP_NAME VARCHAR2 (200)
    PRIMARY KEY(GROUP_NAME)
);
/
CREATE TABLE ITEMS_GROUPS (
	ITEM_NAME VARCHAR2 (200) NOT NULL,
	GROUP_NAME VARCHAR2 (200),
	SENARIO_NAME VARCHAR2 (200) NOT NULL,
    PRIMARY KEY(ITEM_NAME,GROUP_NAME,SENARIO_NAME),
	CONSTRAINT fk_ITEM FOREIGN KEY (ITEM_NAME) REFERENCES ITEMS(ITEM_NAME),
	CONSTRAINT fk_GROUP FOREIGN KEY (GROUP_NAME) REFERENCES GROUPS(GROUP_NAME),
	CONSTRAINT fk_SENARIO FOREIGN KEY (SENARIO_NAME) REFERENCES SENARIOES(SENARIO_NAME)
);
/
CREATE TABLE INPUT_FORMAT (
    EXP_ID NUMBER NOT NULL,
	ITEM_NAME VARCHAR2 (200) NOT NULL,
    PRIMARY KEY(EXP_ID,ITEM_NAME),
	CONSTRAINT fk_INPUT_ITEM FOREIGN KEY (ITEM_NAME) REFERENCES ITEMS(ITEM_NAME),
	CONSTRAINT fk_INPUT_EXP FOREIGN KEY (EXP_ID) REFERENCES Experiments(EXP_ID)
);
/
CREATE TABLE BLACKLIST (
    PARTICIPANT_ID NUMBER NOT NULL,
	PRIMARY KEY(PARTICIPANT_ID)
);
